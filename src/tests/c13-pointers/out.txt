#include <stdio.h>

int
main(int argc, char **argv) {
    char *filename = *(argv + 1);
    printf("filename=%s\n", filename);
    char *a = "hello";
    char **ap = &a;
    printf("a=%p\n", a);
    printf("a=%s\n", a);
    *a = 'y';
    *(a + 1) = 'i';
    *(*ap + 4) = 'i';
    printf("a=%s\n", a);
    return 0;
}
